# deploy

# Linux enviroment

  1. add user

     useradd -d /home/xx -m xxx

 d: user directory

 m: create if not exist

 - delete user if you need
 
     deluser -r xxx

  2. add sudo permissions

     vim /etc/sudoers

 xxx ALL=(ALL) ALL

  3. add to sudo group

     sudo usermod -aG groupName userName

 sudo usermod -aG sudo xxx

- remove user from group

     gpasswd -d xxx sudo

- check user's group

     id xxx

  4. change password

     passwd xxx

  5. change user

     su xxx

# zsh & on-my-zsh

  1. install zsh

     sudo apt-get install zsh

  2. change your bash to zsh

     chsh -s /bin/zsh

  3. change bash to zsh

     sudo vim /etc/passwd

  4. install oh-my-zsh (choose 1 of 2)

     sh -c "$(wget https://raw.githubusercontent.com/robbyrussell/oh-my-zsh/master/tools/install.sh -O -)"

 sh -c "$(curl -fsSL https://github.com/robbyrussell/oh-my-zsh/blob/master/tools/install.sh)"

  5. chagne the default

     sudo vim /etc/passwd
 username:x:1001:1001::/home/username:/bin/zsh

# docker & docker-compose

  1. check os version

     cat /proc/version

  2. install docker

     curl -fsSL https://get.docker.com | bash -s docker --mirror Aliyun

  3. change docker mirror

     sudo vim  /etc/docker/daemon.json
 {"registry-mirrors": ["http://hub-mirror.c.163.com"]}
 sudo systemctl restart docker.service

  4. install docker-compose

     sudo curl -L "https://github.com/docker/compose/releases/download/1.24.1/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose

 sudo chmod +x /usr/local/bin/docker-compose

 docker-compose --version

# Java

  1. Downloads

     https://www.oracle.com/technetwork/java/javase/downloads/jdk8-downloads-2133151.html

  2. Transfre

     scp -P 22 ./jdk-8u221-linux-x64.tar.gz root@47.101.146.110:/home/viber

  3. SetUp

     sudo mkdir /usr/lib/jvm
 sudo tar -zxvf jdk-xx.tar.gz -C /usr/lib/jvm
 sudo vim ~/.bashrc
 export JAVA_HOME=/usr/lib/jvm/jdk1.8.x_x_x
 export JRE_HOME=${JAVA_HOME}/jre
 export CLASSPATH=.:${JAVA_HOME}/lib:${JRE_HOME}/lib
 export PATH=${JAVA_HOME}/bin:$PATH
 sudo update-alternatives –install /usr/bin/java java /usr/lib/jvm/jdk1.7.0_60/bin/java 3

# Maven

1.Downloads

https://maven.apache.org/download.cgi

2.Transfer

3.SetUp

sudo mkdir /usr/local/maven
sudo tar -zxvf maven-xx.tar.gz -C /usr/local/maven

export M2_HOME=/usr/local/apache-maven-x.x.x
export PATH=${M2_HOME}/bin:$PATH

4.Change mirror

<mirror>
    <id>alimaven</id>
    <name>aliyun maven</name>
    <url>http://maven.aliyun.com/nexus/content/groups/public/</url>
    <mirrorOf>central</mirrorOf>
</mirror>
